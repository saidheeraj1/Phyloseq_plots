> GP.chl = subset_taxa(physeq1)

> GP.chl.r = rarefy_even_depth(GP.chl)

> gpsfb.wUF = distance(GP.chl.r, "unifrac", weighted = TRUE)
> gpsfb.NMDS = ordinate(GP.chl.r, "CCA", gpsfb.wUF)
> library("ggplot2")
